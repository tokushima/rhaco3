アクションで発生したエラーを確認する

Flowで実行されたアクションで発生したExceptionをログに出力する
\rhaco\Conf::setで設定する
<?php
\rhaco\Conf::set([
  'rhaco.Log'=>[
    'level'=>'warn', // ログレベル
    'file'=>dirname(__DIR__).'/work/debug.log',  // ログファイルの出力先
  ],
  'rhaco.Flow'=>[
    'exception_log_level'=>'warn', // Exception発生時に出力するエラーレベル
    'exception_log_ignore'=>[ // 出力しない正規表現パターン
      'Unauthorized',
      '.+ require',
    ]
  ],
]);
